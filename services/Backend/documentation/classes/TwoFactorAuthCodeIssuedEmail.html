import { HtmlEmail } from "@/modules/mail/templates/html/HtmlEmail";
import { IEmailTemplate } from "@/modules/mail/templates/IEmailTemplate";

export class TwoFactorAuthCodeIssuedEmail extends HtmlEmail implements IEmailTemplate {
    public constructor(
        private readonly appName: string,
        private readonly appUrl: string,
        private readonly code: string
    ) {
        super();
    }
    public getSubject(): string {
        return "Your 2FA code.";
    }

    protected getRawHtml() {
        return html;
    }

    public getVariables() {
        return {
            appName: this.appName,
            appUrl: this.appUrl,
            code: this.code,
            currentYear: new Date().getFullYear(),
        };
    }
}
